4. Reverse a Number
num = 1234
reversed = 0

while num != 0
  digit = num % 10
  reversed *= 10
  reversed += digit
  num /= 10
end

puts reversed

5. Check if a Number is Prime
num = 13
is_prime = true
i = 2

if num <= 1
  is_prime = false
else
  while i < num
    if num % i == 0
      is_prime = false
    end
    i += 1
  end
end

puts is_prime

6. Check if a Number is Palindrome
num = 121
original_num = num
reversed = 0
is_palindrome = false

while num != 0
  digit = num % 10
  reversed = reversed * 10 + digit
  num /= 10
end

if original_num == reversed
  is_palindrome = true
end

puts is_palindrome

7. Find the Largest Digit in a Number
num = 3947
largest_digit = 0

while num != 0
  digit = num % 10
  if digit > largest_digit
    largest_digit = digit
  end
  num /= 10
end

puts largest_digit
